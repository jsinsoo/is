6. 학습할 코드 1개

라이브러리를 활용한 교실 환경 데이터 분석 프로그램

from statistics import mean, median
from collections import Counter


def validate_readings(readings):
    """0 이상인 수치 데이터만 반환한다."""
    return [
        value for value in readings
        if isinstance(value, (int, float)) and value >= 0
    ]


def analyze_readings(readings):
    """statistics 라이브러리로 핵심 통계값을 계산한다."""
    valid = validate_readings(readings)

    if not valid:
        return None

    levels = [
        "좋음" if value < 800
        else "주의" if value < 1000
        else "환기 필요"
        for value in valid
    ]

    return {
        "count": len(valid),
        "mean": round(mean(valid), 1),
        "median": median(valid),
        "maximum": max(valid),
        "level_counts": Counter(levels)
    }


def print_report(result):
    """분석 결과를 사용자가 이해하기 쉽게 출력한다."""
    if result is None:
        print("분석할 수 있는 데이터가 없습니다.")
        return

    print("유효 측정 수:", result["count"])
    print("평균 농도:", result["mean"], "ppm")
    print("중앙값:", result["median"], "ppm")
    print("최댓값:", result["maximum"], "ppm")

    for level, count in result["level_counts"].items():
        print(level, ":", count, "회")


co2_readings = [620, 710, 850, 930, 1080, 1250, -1, "오류"]

analysis = analyze_readings(co2_readings)
print_report(analysis)
