def is_prime(n):
    if n < 2:
        return False

    divisor = 2
    while divisor * divisor <= n:
        if n % divisor == 0:
            return False
        divisor += 1

    return True

for number in range(2, 30):
    if is_prime(number):
        print(number, end=" ")

예제 8

관찰과 주제

신입생과 방문자가 특별실을 찾기 어려운 문제를 관찰합니다.

주제: 교내 장소를 그래프로 모델링한 최소 이동 경로 안내

from collections import deque

graph = {
    "교실": ["복도"],
    "복도": ["교실", "계단", "도서관"],
    "계단": ["복도", "과학실"],
    "도서관": ["복도", "과학실"],
    "과학실": ["계단", "도서관"]
}

def shortest_path(start, target):
    queue = deque([(start, [start])])
    visited = {start}

    while queue:
        place, path = queue.popleft()
        if place == target:
            return path

        for next_place in graph[place]:
            if next_place not in visited:
                visited.add(next_place)
                queue.append((next_place, path + [next_place]))

print(shortest_path("교실", "과학실"))

records = [
    (30, 8), (45, 9), (60, 7),
    (90, 5), (40, 8)

average_focus = sum(focus for _, focus in records) / len(records)

Ⅶ. 수업 끝 멘트

시는 길가의 작은 꽃 한 송이에서 우주를 발견하고, 소설은 평범한 하루 속에서 한 사람의 깊은 이야기를 찾아냅니다. 희곡은 사소한 대화에서 갈등의 씨앗을 발견하고, 연극은 빈 무대 위에 새로운 세계를 세웁니다. 음악도 일상의 작은 소리를 리듬과 선율로 바꾸어 희망을 들려줍니다.

여러분의 프로젝트도 거대한 기술에서 시작할 필요는 없습니다. 교실의 온도, 복도의 이동, 도서관의 선택, 자신의 공부 기록처럼 가까운 곳에서 충분히 의미 있는 문제를 찾을 수 있습니다. 중요한 것은 문제를 진심으로 바라보고, 데이터를 정직하게 다루며, 결과가 누구에게 어떤 도움을 줄 수 있는지 생각하는 일입니다.

그리고 프로젝트 주제를 “모든 학교 문제를 인공지능으로 해결하기”로 정하면 범위가 우주보다 넓어져 제출 시각까지 함수가 반환되지 않을 수 있습니다. 오늘 발견한 작은 질문 하나가 여러분만의 특별한 연구와 큰 꿈으로 성장하기를 바랍니다.
