def sum_loop(n):
    total = 0

    for number in range(1, n + 1):
        total += number

    return total

재귀함수

def sum_recursive(n):
    if n == 1:
        return 1

예제 8

이진 탐색

분할 정복탐색

반복문

def binary_search_loop(data, target):
    left = 0
    right = len(data) - 1

    while left <= right:
        middle = (left + right) // 2

        if data[middle] == target:
            return middle
        elif data[middle] < target:
            left = middle + 1
        else:
            right = middle - 1

    return -1

재귀함수

def binary_search_recursive(
    data, target, left, right
):
    if left > right:
        return -1

    middle = (left + right) // 2

    if data[middle] == target:
        return middle
    if data[middle] < target:
        return binary_search_recursive(
            data, target, middle + 1, right
        )

    return binary_search_recursive(
        data, target, left, middle - 1
    )
알고리즘 구조
이진 탐색은 탐색 범위를 절반으로 줄이는 분할 정복 문제이므로 재귀 구조가 자연스럽습니다. 반복문은 함수 호출 없이 같은 효율을 얻습니다.

예제 10

import time

def factorial_loop(n):
    result = 1
    for number in range(1, n + 1):
        result *= number
    return result

def factorial_recursive(n):
    if n == 0:
        return 1
    return n * factorial_recursive(n - 1)

n = 900

start = time.perf_counter()
loop_result = factorial_loop(n)
loop_time = time.perf_counter() - start

start = time.perf_counter()
recursive_result = factorial_recursive(n)
recursive_time = time.perf_counter() - start

print("결과 일치:", loop_result == recursive_result)
print(f"반복문 시간: {loop_time:.8f}초")
print(f"재귀 시간: {recursive_time:.8f}초")

Ⅶ. 수업 끝 멘트

“시는 같은 마음을 운율로 반복하고, 소설은 한 인물의 선택을 여러 장면에서 되돌아보게 합니다. 희곡은 대사의 반복 속에서 의미를 바꾸고, 연극은 매일 같은 막을 올리면서도 어제보다 깊은 장면을 만듭니다. 음악의 후렴도 반복되지만 같은 감동에 머물지 않습니다. 여러분의 꿈에도 반복문이 필요할 때가 있고 재귀함수가 필요할 때가 있습니다. 매일 조금씩 실천할 일은 반복문처럼, 스스로에게 ‘왜?’를 다시 묻는 성찰은 재귀처럼 이어 가세요. 다만 고민을 무한 재귀로 돌리면 마음에 RecursionError가 날 수 있으니 친구와 선생님이라는 종료 조건도 호출해 보세요.”

Ⅷ. 형성평가 OX 퀴즈 10개
