예제 5

Stack 클래스로 연산 통합

클래스캡슐화
class Stack:
    def __init__(self):
        self.items = []

    def push(self, value):
        self.items.append(value)

    def pop(self):
        if self.is_empty():
            return None
        return self.items.pop()

    def peek(self):
        if self.is_empty():
            return None
        return self.items[-1]

    def is_empty(self):
        return len(self.items) == 0

    def size(self):
        return len(self.items)

stack = Stack()
stack.push(100)
stack.push(200)

print("peek:", stack.peek())
print("pop:", stack.pop())
print("크기:", stack.size())
장점
스택 자료와 연산을 하나의 클래스로 묶어 여러 스택 객체를 독립적으로 만들 수 있습니다.

예제 7

문서 편집 실행 취소

실행 취소상태 저장
text = ""
history = []

def write(new_text):
    global text

    history.append(text)
    text += new_text
    print("현재 문서:", text)

def undo():
    global text

    if not history:
        print("취소할 작업이 없습니다.")
        return

    text = history.pop()
    print("실행 취소:", text)

write("안녕")
write("하세요")
write("!")
undo()
undo()

def reverse_text(text):
    stack = []

    for character in text:
        stack.append(character)

    result = ""

    while stack:
        result += stack.pop()

    return result

def check_parentheses(expression):
    stack = []

    for character in expression:
        if character == "(":
            stack.append(character)

        elif character == ")":
            if not stack:
                return False
            stack.pop()

    return len(stack) == 0

tests = [
    "(a+b)*(c-d)",
    "((a+b)",
    "a+b)",
    "(a+(b*c))"
]

for test in tests:
    print(test, "→", check_parentheses(test))

명령어 기반 스택 프로그램

종합 실습push·pop·peek
stack = []

def push(value):
    stack.append(value)

def pop():
    if not stack:
        return "EMPTY"
    return stack.pop()

def peek():
    if not stack:
        return "EMPTY"
    return stack[-1]

while True:
    command = input("명령: ").split()

    if command[0] == "push":
        push(command[1])
        print("스택:", stack)

    elif command[0] == "pop":
        print("삭제:", pop())

    elif command[0] == "peek":
        print("맨 위:", peek())

    elif command[0] == "size":
        print("크기:", len(stack))

    elif command[0] == "quit":
        break

    else:
        print("알 수 없는 명령입니다.")
