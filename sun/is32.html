    if not values:
        raise ValueError(

예제 8

unittest로 정상·경계·예외 테스트 자동화

unittestassertRaises
import unittest

def classify_score(score):
    if not isinstance(score, (int, float)):
        raise TypeError(
            "점수는 숫자여야 합니다."
        )

    if not 0 <= score <= 100:
        raise ValueError(
            "점수 범위는 0~100입니다."
        )

    return "합격" if score >= 60 else "불합격"

class TestClassifyScore(unittest.TestCase):
    def test_normal(self):
        self.assertEqual(
            classify_score(75),
            "합격"
        )

    def test_boundaries(self):
        self.assertEqual(
            classify_score(60),
            "합격"
        )
        self.assertEqual(
            classify_score(59),
            "불합격"
        )
        self.assertEqual(
            classify_score(0),
            "불합격"
        )
        self.assertEqual(
            classify_score(100),
            "합격"
        )

    def test_invalid_type(self):
        with self.assertRaises(TypeError):
            classify_score("75")

    def test_out_of_range(self):
        with self.assertRaises(ValueError):
            classify_score(101)

if __name__ == "__main__":
    unittest.main()
자동화
코드를 수정할 때마다 테스트를 다시 실행하면 이전 기능이 깨졌는지 빠르게 확인할 수 있습니다.

def calculate_discount(price, rate):
    if not isinstance(price, (int, float)):
        raise TypeError(
            "가격은 숫자여야 합니다."
        )

    if not isinstance(rate, (int, float)):
        raise TypeError(
            "할인율은 숫자여야 합니다."
        )

    if price < 0:
        raise ValueError(
            "가격은 0 이상이어야 합니다."
        )

    if not 0 <= rate <= 100:
        raise ValueError(
            "할인율은 0~100이어야 합니다."
        )

    return price * (1 - rate / 100)

test_cases = [
    {
        "name": "정상 할인",
        "args": (10000, 20),
        "expected": 8000
    },
    {
        "name": "경계 할인 0%",
        "args": (10000, 0),
        "expected": 10000
    },
    {
        "name": "경계 할인 100%",
        "args": (10000, 100),
        "expected": 0
    },
    {
        "name": "가격 0원",
        "args": (0, 50),
        "expected": 0
    },
    {
        "name": "음수 가격",
        "args": (-1000, 10),
        "exception": ValueError
    },
    {
        "name": "할인율 초과",
        "args": (10000, 120),
        "exception": ValueError
    },
    {
        "name": "문자 가격",
        "args": ("10000", 20),
        "exception": TypeError
    }
]

passed = 0

for case in test_cases:
    try:
        result = calculate_discount(
            *case["args"]
        )

        if "exception" in case:
            print(
                "FAIL",
                case["name"],
                "예외가 발생하지 않음"
            )
            continue

        if result == case["expected"]:
            print("PASS", case["name"])
            passed += 1
        else:
            print(
                "FAIL",
                case["name"],
                result,
                case["expected"]
            )

    except Exception as error:
        expected_error = case.get(
            "exception"
        )

        if (
            expected_error
            and isinstance(error, expected_error)
        ):
            print("PASS", case["name"])
            passed += 1
        else:
            print(
                "FAIL",
                case["name"],
                type(error).__name__,
                error
            )

print(
    f"통과: {passed}/{len(test_cases)}"
)

Ⅶ. 수업 끝 멘트

시는 지워진 단어와 다시 고친 한 줄 속에서 더 깊어지고, 소설은 인물의 실패가 다음 장면의 성장을 만듭니다. 희곡은 갈등과 실수가 있어야 변화가 시작되고, 연극은 리허설에서 오류를 많이 발견할수록 본 공연이 단단해집니다. 음악도 틀린 음을 찾아 다시 맞추는 과정에서 아름다운 화음이 완성됩니다.

여러분의 꿈도 처음부터 오류 없는 프로그램처럼 실행되지는 않습니다. 중요한 것은 실패를 숨기지 않고 어떤 조건에서 흔들렸는지 살피며, 작은 수정과 반복 연습으로 더 안정적인 자신을 만들어 가는 일입니다.

버그를 발견한 뒤 컴퓨터를 껐다 켜서 사라졌다고 보고하면 오류 수정이 아니라 오류와의 숨바꼭질입니다. 오늘 배운 재현·수정·기록의 태도가 코드와 삶의 어려움을 차분히 해결하는 힘이 되기를 바랍니다.
