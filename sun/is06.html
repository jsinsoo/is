def hanoi(n, start, target, helper):
    if n == 1:
        print(f"{start} → {target}")
        return

    hanoi(n - 1, start, helper, target)
    print(f"{start} → {target}")
    hanoi(n - 1, helper, target, start)

hanoi(3, "A", "C", "B")

  3. 보조 기둥의 n-1개를 목표 기둥으로 옮깁니다.

예제 4

깊이 우선 방식의 미로 탐색

DFS방문 표시미로
maze = [
    [0, 1, 0, 0],
    [0, 0, 0, 1],
    [1, 0, 1, 0],
    [0, 0, 0, 0]
]

rows = len(maze)
cols = len(maze[0])
visited = [[False] * cols for _ in range(rows)]

def solve(row, col):
    if row < 0 or row >= rows:
        return False
    if col < 0 or col >= cols:
        return False
    if maze[row][col] == 1 or visited[row][col]:
        return False

    if row == rows - 1 and col == cols - 1:
        return True

    visited[row][col] = True

    return (
        solve(row + 1, col)
        or solve(row - 1, col)
        or solve(row, col + 1)
        or solve(row, col - 1)
    )

print("탈출 가능:", solve(0, 0))
종료 조건
배열 밖, 벽, 방문한 칸이면 실패하고, 출구에 도착하면 성공을 반환합니다.

예제 5

미로 경로 저장과 복원

백트래킹경로 복원
maze = [
    [0, 1, 0, 0],
    [0, 0, 0, 1],
    [1, 0, 1, 0],
    [0, 0, 0, 0]
]

rows = len(maze)
cols = len(maze[0])
visited = set()
path = []

def find_path(row, col):
    if not (0 <= row < rows and 0 <= col < cols):
        return False
    if maze[row][col] == 1 or (row, col) in visited:
        return False

    visited.add((row, col))
    path.append((row, col))

    if (row, col) == (rows - 1, cols - 1):
        return True

    directions = [(1,0), (-1,0), (0,1), (0,-1)]

    for dr, dc in directions:
        if find_path(row + dr, col + dc):
            return True

    path.pop()
    return False

if find_path(0, 0):
    print("경로:", path)
else:
    print("경로 없음")
백트래킹
현재 선택이 출구로 이어지지 않으면 path.pop()으로 마지막 위치를 제거하고 이전 상태로 돌아갑니다.

예제 6

미로의 모든 경로 찾기

모든 해방문 해제
maze = [
    [0, 0, 0],
    [0, 1, 0],
    [0, 0, 0]
]

rows = len(maze)
cols = len(maze[0])
visited = set()
all_paths = []

def find_all_paths(row, col, path):
    if not (0 <= row < rows and 0 <= col < cols):
        return
    if maze[row][col] == 1 or (row, col) in visited:
        return

    visited.add((row, col))
    path.append((row, col))

    if (row, col) == (rows - 1, cols - 1):
        all_paths.append(path.copy())
    else:
        for dr, dc in [(1,0), (-1,0), (0,1), (0,-1)]:
            find_all_paths(row + dr, col + dc, path)

    path.pop()
    visited.remove((row, col))

find_all_paths(0, 0, [])

print("경로 수:", len(all_paths))
for path in all_paths:
    print(path)

def permutations(items, selected):
    if len(selected) == len(items):
        print(selected)
        return

    for item in items:
        if item not in selected:
            selected.append(item)
            permutations(items, selected)
            selected.pop()

예제 8

조합시작 인덱스
def combinations(items, r, start=0, selected=None):
    if selected is None:
        selected = []

    if len(selected) == r:
        print(selected)
        return

    for index in range(start, len(items)):
        selected.append(items[index])
        combinations(items, r, index + 1, selected)
        selected.pop()

combinations(["A", "B", "C", "D"], 2)

포함·제외 방식의 부분집합 생성

부분집합이진 선택
def subsets(items, index=0, selected=None):
    if selected is None:
        selected = []

    if index == len(items):
        print(selected)
        return

    # 현재 원소를 포함
    selected.append(items[index])
    subsets(items, index + 1, selected)

    # 현재 원소를 제외
    selected.pop()
    subsets(items, index + 1, selected)

subsets([1, 2, 3])

  2. 현재 원소를 부분집합에서 제외한다.
