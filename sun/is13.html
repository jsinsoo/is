예제 4

간선 연산O(1) 확인
nodes = ["A", "B", "C", "D"]
index = {node: i for i, node in enumerate(nodes)}
matrix = [[0] * 4 for _ in range(4)]

def add_edge(a, b):
    i, j = index[a], index[b]
    matrix[i][j] = 1
    matrix[j][i] = 1

def remove_edge(a, b):
    i, j = index[a], index[b]
    matrix[i][j] = 0
    matrix[j][i] = 0

def has_edge(a, b):
    return matrix[index[a]][index[b]] == 1

add_edge("A", "B")
add_edge("A", "C")

print(has_edge("A", "C"))
remove_edge("A", "C")
print(has_edge("A", "C"))
장점
두 노드의 인덱스를 알면 행렬의 한 칸만 확인하여 연결 여부를 빠르게 판정할 수 있습니다.

예제 6

인접리스트의 간선 삽입·삭제·탐색

리스트 연산이웃 탐색
graph = {
    "A": [],
    "B": [],
    "C": [],
    "D": []
}

def add_edge(a, b):
    if b not in graph[a]:
        graph[a].append(b)

    if a not in graph[b]:
        graph[b].append(a)

def remove_edge(a, b):
    if b in graph[a]:
        graph[a].remove(b)

    if a in graph[b]:
        graph[b].remove(a)

def has_edge(a, b):
    return b in graph[a]

add_edge("A", "B")
add_edge("A", "C")

print(graph)
print(has_edge("A", "C"))

remove_edge("A", "C")
print(graph)

]

예제 10

교실 협업 관계 종합 분석

종합 프로그램차수 분석
students = [
    "민준", "서연", "지우",
    "하준", "예린", "도윤"
]

collaborations = [
    ("민준", "서연"),
    ("민준", "지우"),
    ("민준", "하준"),
    ("서연", "지우"),
    ("지우", "예린"),
    ("하준", "도윤")
]

# 인접리스트 생성
graph = {student: [] for student in students}

for a, b in collaborations:
    graph[a].append(b)
    graph[b].append(a)

# 차수 계산
degrees = {
    student: len(neighbors)
    for student, neighbors in graph.items()
}

# 가장 연결이 많은 학생
most_connected = max(
    degrees,
    key=degrees.get
)

print("[인접리스트]")
for student, neighbors in graph.items():
    print(student, ":", neighbors)

print("\n[차수]")
for student, degree in degrees.items():
    print(student, ":", degree)

print(
    "\n협업 연결이 가장 많은 학생:",
    most_connected,
    degrees[most_connected]
)

# 인접행렬 생성
index = {
    student: i
    for i, student in enumerate(students)
}

n = len(students)
matrix = [[0] * n for _ in range(n)]

for a, b in collaborations:
    i, j = index[a], index[b]
    matrix[i][j] = 1
    matrix[j][i] = 1

print("\n[인접행렬]")
for row in matrix:
    print(row)
분석 관점인접행렬인접리스트

가중치 그래프와 간선 비용 분석

가중치도로 거리인접리스트
nodes = ["학교", "도서관", "공원", "정류장"]

roads = [
    ("학교", "도서관", 4),
    ("학교", "공원", 7),
    ("도서관", "정류장", 3),
    ("공원", "정류장", 2)
]

graph = {node: [] for node in nodes}

for start, end, distance in roads:
    graph[start].append((end, distance))
    graph[end].append((start, distance))

for node, neighbors in graph.items():
    print(node, ":", neighbors)

total_distance = sum(
    distance
    for _, _, distance in roads
)

average_distance = total_distance / len(roads)

print("전체 도로 거리:", total_distance)
print("평균 도로 거리:", average_distance)

long_roads = [
    (start, end, distance)
    for start, end, distance in roads
    if distance >= 5
]

print("5 이상 도로:", long_roads)

Ⅶ. 수업 끝 멘트

“시는 한 단어와 다른 단어를 연결해 마음의 길을 만들고, 소설은 인물과 사건의 간선이 얽혀 하나의 세계를 이룹니다. 희곡은 배우들의 관계가 달라질 때 장면의 의미가 바뀌고, 연극은 무대 위 모든 연결이 조화를 이룰 때 살아납니다. 음악도 음표 하나보다 음과 음 사이의 관계가 더 큰 감동을 만듭니다. 여러분의 꿈도 혼자 떨어진 노드가 아니라 가족·친구·선생님과 이어진 그래프 속에서 자랍니다. 다만 관계를 너무 많이 추가하면 인접행렬이 복잡해질 수 있으니 진심 없는 간선은 가끔 remove_edge 해도 좋습니다. 좋은 연결을 만들고 서로의 가능성을 이어 주는 사람이 되길 바랍니다.”
