실생활 응용자동 번호
from collections import deque

waiting = deque()
next_number = 1

def issue_ticket(name):
    global next_number

    waiting.append((next_number, name))
    print(f"{name}님의 번호는 {next_number}번입니다.")
    next_number += 1

def call_next():
    if not waiting:
        print("대기자가 없습니다.")
        return

    number, name = waiting.popleft()
    print(f"{number}번 {name}님, 입장하세요.")

issue_ticket("지우")

frontrear한계 분석
capacity = 5
queue = [None] * capacity
front = 0
rear = 0

def enqueue(value):
    global rear

    if rear == capacity:
        print("rear가 끝에 도달했습니다.")
        return

    queue[rear] = value
    rear += 1

def dequeue():
    global front

    if front == rear:
        return None

    value = queue[front]
    queue[front] = None
    front += 1
    return value

for value in [10, 20, 30, 40, 50]:

예제 8

원형 큐 클래스 직접 구현

원형 배열클래스
class CircularQueue:
    def __init__(self, capacity):
        self.capacity = capacity
        self.data = [None] * capacity
        self.front = 0
        self.rear = 0
        self.count = 0

    def is_empty(self):
        return self.count == 0

    def is_full(self):
        return self.count == self.capacity

    def enqueue(self, value):
        if self.is_full():
            raise OverflowError("큐가 가득 찼습니다.")

        self.data[self.rear] = value
        self.rear = (self.rear + 1) % self.capacity
        self.count += 1

    def dequeue(self):
        if self.is_empty():
            raise IndexError("큐가 비어 있습니다.")

        value = self.data[self.front]
        self.data[self.front] = None
        self.front = (self.front + 1) % self.capacity
        self.count -= 1
        return value

    def peek(self):
        if self.is_empty():
            return None
        return self.data[self.front]

queue = CircularQueue(4)

queue.enqueue("A")
queue.enqueue("B")
queue.enqueue("C")

print(queue.dequeue())
queue.enqueue("D")
queue.enqueue("E")

print(queue.data)
print("front:", queue.front)
print("rear:", queue.rear)
count의 역할
front와 rear가 같은 경우 빈 상태인지 가득 찬 상태인지 구분하기 위해 저장된 자료 수를 별도로 관리합니다.

class CircularQueue:
    def __init__(self, capacity):
        self.data = [None] * capacity
        self.capacity = capacity
        self.front = 0
        self.rear = 0
        self.count = 0

    def show(self, action):
        print(
            f"{action:12} | data={self.data} "
            f"| front={self.front} "
            f"| rear={self.rear} "
            f"| count={self.count}"
        )

    def enqueue(self, value):
        if self.count == self.capacity:
            self.show("enqueue 실패")
            return

        self.data[self.rear] = value
        self.rear = (self.rear + 1) % self.capacity
        self.count += 1
        self.show(f"enqueue {value}")

    def dequeue(self):
        if self.count == 0:
            self.show("dequeue 실패")
            return None

        value = self.data[self.front]
        self.data[self.front] = None
        self.front = (self.front + 1) % self.capacity
        self.count -= 1
        self.show(f"dequeue {value}")
        return value

종합 프로그램작업 처리
class PrintQueue:
    def __init__(self, capacity=5):
        self.capacity = capacity
        self.data = [None] * capacity
        self.front = 0
        self.rear = 0
        self.count = 0

    def add_job(self, document):
        if self.count == self.capacity:
            print("대기열이 가득 찼습니다.")
            return

        self.data[self.rear] = document
        self.rear = (self.rear + 1) % self.capacity
        self.count += 1
        print("추가:", document)

    def print_next(self):
        if self.count == 0:
            print("인쇄할 문서가 없습니다.")
            return

        document = self.data[self.front]
        self.data[self.front] = None
        self.front = (self.front + 1) % self.capacity
        self.count -= 1
        print("인쇄:", document)

    def show_jobs(self):
        jobs = []

        for offset in range(self.count):
            index = (self.front + offset) % self.capacity
            jobs.append(self.data[index])

        print("대기 문서:", jobs)

printer = PrintQueue(4)

printer.add_job("수업자료.pdf")
printer.add_job("평가계획.hwp")
printer.add_job("학생명단.xlsx")

printer.print_next()
printer.add_job("알고리즘.pdf")
printer.add_job("과제안내.txt")

printer.show_jobs()

class CircularQueue:
    def __init__(self, capacity=5):
        self.capacity = capacity
        self.data = [None] * capacity
        self.front = 0
        self.rear = 0
        self.count = 0

    def enqueue(self, value):
        if self.count == self.capacity:
            return "FULL"

        self.data[self.rear] = value
        self.rear = (self.rear + 1) % self.capacity
        self.count += 1
        return value

    def dequeue(self):
        if self.count == 0:
            return "EMPTY"

        value = self.data[self.front]
        self.data[self.front] = None
        self.front = (self.front + 1) % self.capacity
        self.count -= 1
        return value

    def peek(self):
        if self.count == 0:
            return "EMPTY"
        return self.data[self.front]

    def show(self):
        result = []

        for offset in range(self.count):
            index = (self.front + offset) % self.capacity
            result.append(self.data[index])

        return result

queue = CircularQueue(5)

while True:
    command = input("명령: ").split()

    if command[0] == "enqueue":
        if len(command) == 1:
            print("추가할 값을 입력하세요.")
            continue
        print("추가:", queue.enqueue(command[1]))

    elif command[0] == "dequeue":
        print("삭제:", queue.dequeue())

    elif command[0] == "peek":
        print("맨 앞:", queue.peek())

    elif command[0] == "show":
        print("논리적 큐:", queue.show())

    elif command[0] == "state":
        print(
            "배열:", queue.data,
            "front:", queue.front,
            "rear:", queue.rear,
            "count:", queue.count
        )

    elif command[0] == "quit":
        break

    else:
        print("알 수 없는 명령입니다.")
명령기능
enqueue 값rear 위치에 자료를 추가합니다.
dequeuefront 위치 자료를 제거하고 반환합니다.
peek맨 앞 자료를 삭제하지 않고 확인합니다.
showfront부터 시작하는 논리적 큐 순서를 출력합니다.
state내부 배열·front·rear·count 상태를 출력합니다.
quit프로그램을 종료합니다.
상세 설명
원형 큐의 내부 배열 순서와 실제 처리 순서는 다를 수 있습니다. show()는 front부터 count개의 원소를 나머지 연산으로 순회하여 논리적인 FIFO 순서를 복원합니다.

“시는 첫 구절이 다음 구절에게 자리를 내어 주고, 소설은 사건이 일어난 순서 속에서 인물의 삶을 이해하게 합니다. 희곡은 배우가 자기 차례를 기다릴 때 장면이 흐르고, 연극은 무대 뒤의 대기 순서까지 지켜질 때 완성됩니다. 음악도 먼저 시작한 선율이 다음 선율에게 자연스럽게 이어집니다. 여러분의 꿈에는 기다림도 필요합니다. 아직 차례가 오지 않았다고 실패한 것이 아니라 여러분의 데이터가 큐에서 안전하게 기다리는 중일 수 있습니다. 다만 숙제를 마감일 뒤까지 enqueue만 하고 dequeue하지 않으면 큐가 가득 찰 수 있습니다. 오늘의 작은 과제부터 하나씩 처리하며 꿈의 front를 앞으로 움직여 봅시다.”
