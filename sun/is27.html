예제 2

값 탐색: 선형 탐색과 이진 탐색

O(n)O(log n)
def linear_search(data, target):
    for index, value in enumerate(data):
        if value == target:
            return index
    return -1

def binary_search(data, target):
    left = 0
    right = len(data) - 1

    while left <= right:
        middle = (left + right) // 2

        if data[middle] == target:
            return middle

        if data[middle] < target:
            left = middle + 1
        else:
            right = middle - 1

    return -1

data = [2, 5, 8, 12, 16, 23, 38]

print(linear_search(data, 16))
print(binary_search(data, 16))
적용 조건
이진 탐색은 자료가 정렬되어 있어야 합니다. 정렬되지 않은 일회성 자료라면 선형 탐색이 더 단순하고 적절할 수 있습니다.

예제 3

두 수의 합: 모든 쌍과 딕셔너리

O(n²)평균 O(n)
def two_sum_nested(data, target):
    for i in range(len(data)):
        for j in range(i + 1, len(data)):
            if data[i] + data[j] == target:
                return i, j

    return None
def two_sum_hash(data, target):
    positions = {}

    for index, value in enumerate(data):
        needed = target - value

        if needed in positions:
            return positions[needed], index

        positions[value] = index

    return None
numbers = [2, 7, 11, 15]
target = 9

print(two_sum_nested(numbers, target))
print(two_sum_hash(numbers, target))
비교
딕셔너리 방식은 추가 메모리를 사용하지만 평균적으로 한 번의 순회로 해결합니다. 모든 쌍 방식은 메모리는 적게 쓰지만 비교 횟수가 많습니다.

예제 4

중복 존재 확인: 정렬과 집합

O(n log n)평균 O(n)
def duplicate_by_sort(data):
    ordered = sorted(data)

    for index in range(len(ordered) - 1):
        if ordered[index] == ordered[index + 1]:
            return True

    return False

def duplicate_by_set(data):
    seen = set()

    for value in data:
        if value in seen:
            return True

        seen.add(value)

    return False

numbers = [4, 1, 7, 3, 7]

print(duplicate_by_sort(numbers))
print(duplicate_by_set(numbers))
적용 조건
집합 방식은 빠르지만 해시 가능한 자료만 저장할 수 있습니다. 정렬 방식은 결과 순서도 함께 필요할 때 활용할 수 있습니다.

예제 5

정렬: 선택 정렬과 병합 정렬

O(n²)O(n log n)
def selection_sort(data):
    result = data.copy()

    for start in range(len(result) - 1):
        minimum = start

        for index in range(start + 1, len(result)):
            if result[index] < result[minimum]:
                minimum = index

        result[start], result[minimum] = (
            result[minimum], result[start]
        )

    return result
def merge_sort(data):
    if len(data) <= 1:
        return data

    middle = len(data) // 2
    left = merge_sort(data[:middle])
    right = merge_sort(data[middle:])

    result = []
    i = j = 0

    while i < len(left) and j < len(right):
        if left[i] <= right[j]:
            result.append(left[i])
            i += 1
        else:
            result.append(right[j])
            j += 1

    return result + left[i:] + right[j:]
비교
선택 정렬은 구현이 단순하고 교환 횟수가 적지만 큰 입력에는 느립니다. 병합 정렬은 추가 배열이 필요하지만 큰 입력에서 안정적인 성능을 보입니다.

예제 6

피보나치: 단순 재귀와 동적 계획법

지수 시간O(n)
def fib_recursive(n):
    if n < 2:
        return n

    return (
        fib_recursive(n - 1)
        + fib_recursive(n - 2)
    )

def fib_table(n):
    if n < 2:
        return n

    dp = [0] * (n + 1)
    dp[1] = 1

    for index in range(2, n + 1):
        dp[index] = (
            dp[index - 1]
            + dp[index - 2]
        )

    return dp[n]

for n in range(11):
    print(
        n,
        fib_recursive(n),
        fib_table(n)
    )
정확성
두 알고리즘 모두 같은 점화식을 따르므로 같은 값을 반환합니다. 동적 계획법은 중복 부분 문제를 저장하여 반복 계산을 제거합니다.

예제 7

최단 경로: BFS와 다익스트라

무가중치가중치
from collections import deque

def bfs_distance(graph, start):
    queue = deque([start])
    distance = {start: 0}

    while queue:
        node = queue.popleft()

        for neighbor in graph[node]:
            if neighbor not in distance:
                distance[neighbor] = (
                    distance[node] + 1
                )
                queue.append(neighbor)

    return distance
import heapq

def dijkstra(graph, start):
    distance = {
        node: float("inf")
        for node in graph
    }
    distance[start] = 0

    heap = [(0, start)]

    while heap:
        current, node = heapq.heappop(heap)

        if current > distance[node]:
            continue

        for neighbor, weight in graph[node]:
            new_distance = current + weight

            if new_distance < distance[neighbor]:
                distance[neighbor] = new_distance
                heapq.heappush(
                    heap,
                    (new_distance, neighbor)
                )

    return distance
적용 조건
BFS는 모든 간선 비용이 같을 때 최단 간선 수를 구합니다. 다익스트라는 음수가 아닌 서로 다른 가중치를 처리할 수 있습니다.

예제 8

부분합: 완전 탐색과 백트래킹

2ⁿ가지치기
numbers = [2, 3, 5, 7, 8]
target = 10

def subset_full(index, total):
    if index == len(numbers):
        return total == target

    return (
        subset_full(
            index + 1,
            total + numbers[index]
        )
        or subset_full(
            index + 1,
            total
        )
    )

def subset_pruned(index, total):
    if total == target:
        return True

    if total > target:
        return False

    if index == len(numbers):
        return False

    return (
        subset_pruned(
            index + 1,
            total + numbers[index]
        )
        or subset_pruned(
            index + 1,
            total
        )
    )

print(subset_full(0, 0))
print(subset_pruned(0, 0))
정확성 조건
total > target 가지치기는 모든 숫자가 양수일 때만 안전합니다. 음수가 포함되면 이후 합이 다시 줄어들 수 있습니다.

예제 9

동전 교환: 탐욕법과 동적 계획법

빠른 탐욕최적 보장 DP
def greedy_coin(coins, amount):
    count = 0

    for coin in sorted(
        coins,
        reverse=True
    ):
        count += amount // coin
        amount %= coin

    return count
def dp_coin(coins, amount):
    dp = [float("inf")] * (amount + 1)
    dp[0] = 0

    for value in range(1, amount + 1):
        for coin in coins:
            if coin <= value:
                dp[value] = min(
                    dp[value],
                    dp[value - coin] + 1
                )

    return dp[amount]
coins = [1, 3, 4]
amount = 6

print("탐욕:", greedy_coin(coins, amount))
print("DP:", dp_coin(coins, amount))

def choose_algorithm(problem, condition):
    if problem == "탐색":
        if condition == "정렬됨":
            return "이진 탐색", "O(log n)"
        return "선형 탐색", "O(n)"

    if problem == "최단 경로":
        if condition == "모든 간선 비용 동일":
            return "BFS", "O(V+E)"
        if condition == "음수 없는 가중치":
            return "다익스트라", "O((V+E)log V)"

    if problem == "중복 검사":
        if condition == "추가 메모리 가능":
            return "집합", "평균 O(n)"
        return "정렬 후 인접 비교", "O(n log n)"

    if problem == "부분합":
        if condition == "양수만":
            return "백트래킹+가지치기", "최악 O(2^n)"
        return "완전 탐색 또는 DP", "조건에 따라 다름"

    return "추가 분석 필요", "-"

cases = [
    ("탐색", "정렬됨"),
    ("탐색", "정렬 안 됨"),
    ("최단 경로", "모든 간선 비용 동일"),
    ("최단 경로", "음수 없는 가중치"),
    ("중복 검사", "추가 메모리 가능"),
    ("부분합", "양수만")
]

for problem, condition in cases:
    algorithm, complexity = choose_algorithm(
        problem,
        condition
    )

    print(
        problem,
        condition,
        "→",
        algorithm,
        complexity
    )

비교 설명 예시
“정렬된 자료에서 값을 한 번만 찾는다면 선형 탐색도 사용할 수 있지만, 같은 자료에서 여러 번 탐색한다면 이진 탐색이 유리하다. 선형 탐색은 O(n)이며 정렬 조건이 없고, 이진 탐색은 O(log n)이지만 정렬된 자료가 필요하다. 따라서 사전 정렬 비용과 탐색 반복 횟수를 함께 고려해 선택해야 한다.”

시는 같은 마음을 여러 비유로 표현하고, 소설은 한 사건을 여러 인물의 시선으로 바라봅니다. 희곡은 같은 대사도 배우와 장면에 따라 다른 의미가 되고, 연극은 하나의 이야기를 다양한 연출로 새롭게 만듭니다. 음악도 같은 선율을 피아노와 바이올린이 서로 다른 빛깔로 들려줍니다.

여러분의 꿈도 하나의 정답만 있는 문제가 아닙니다. 누구는 빠른 길을 선택하고, 누구는 더 많은 준비와 메모리를 사용하며, 누구는 단순하지만 오래 지속할 수 있는 방법을 택합니다. 중요한 것은 남의 알고리즘을 따라가는 것이 아니라 자신의 조건을 이해하고 선택의 근거를 만드는 일입니다.

그리고 시험에서 답을 두 개 쓰고 “복수 알고리즘을 설계했습니다”라고 주장하면 채점 기준의 적용 조건을 통과하기 어렵습니다. 오늘 배운 비교의 눈으로 코드뿐 아니라 자신의 시간과 선택도 더 현명하게 설계해 보세요.
