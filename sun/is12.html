예제 5

배식대 1개와 2개 비교

병렬 처리개선 분석
students = [
    ("민준", 0, 3),
    ("서연", 0, 2),
    ("지우", 1, 4),
    ("하준", 2, 2),
    ("예린", 3, 3)
]

def simulate(server_count):
    servers = [0] * server_count
    waiting_times = []

    for name, arrival, service in students:
        server_index = min(
            range(server_count),
            key=lambda index: servers[index]
        )

        start = max(
            arrival,
            servers[server_index]
        )

        waiting = start - arrival
        waiting_times.append(waiting)

        servers[server_index] = start + service

    return sum(waiting_times) / len(waiting_times)

print("배식대 1개:", simulate(1))
print("배식대 2개:", simulate(2))
분석
처리 창구를 늘리면 동시에 여러 학생을 처리할 수 있어 평균 대기시간이 줄어드는지 비교할 수 있습니다.

예제 7

긴 작업이 앞에 있을 때의 영향

작업 순서평균 비교
jobs = [
    ("긴 보고서", 10),
    ("짧은 문서1", 1),
    ("짧은 문서2", 2),
    ("짧은 문서3", 1)
]

def average_waiting(job_list):
    current_time = 0
    waits = []

    for name, service in job_list:
        waits.append(current_time)
        current_time += service

    return sum(waits) / len(waits)

fifo_average = average_waiting(jobs)

short_first = sorted(
    jobs,
    key=lambda job: job[1]
)
short_average = average_waiting(short_first)

print("FIFO 평균:", fifo_average)
print("짧은 작업 우선 평균:", short_average)
윤리·운영 관점
짧은 작업 우선은 평균 대기시간을 줄일 수 있지만 먼저 도착한 작업이 늦어질 수 있습니다. 공정성과 효율성을 함께 고려해야 합니다.

예제 10

종합 대기열 시뮬레이션 함수

종합 프로그램재사용 함수
from statistics import mean, median

def analyze_queue(customers, server_count=1):
    servers = [0] * server_count
    results = []

    for name, arrival, service in customers:
        server_index = min(
            range(server_count),
            key=lambda index: servers[index]
        )

        start = max(
            arrival,
            servers[server_index]
        )

        waiting = start - arrival
        finish = start + service

        servers[server_index] = finish

        results.append({
            "이름": name,
            "도착": arrival,
            "시작": start,
            "종료": finish,
            "대기": waiting,
            "창구": server_index + 1
        })

    waits = [item["대기"] for item in results]

    summary = {
        "평균": mean(waits),
        "중앙값": median(waits),
        "최대": max(waits),
        "대기 0명": waits.count(0)
    }

    return results, summary

students = [
    ("민준", 0, 3),
    ("서연", 0, 2),
    ("지우", 1, 4),
    ("하준", 2, 2),
    ("예린", 5, 3)
]

results, summary = analyze_queue(
    students,
    server_count=2
)

for item in results:
    print(item)

예제 11

도착률·처리율 기반 혼잡도 분석

도착률처리율혼잡도
def analyze_load(arrival_rate, service_rate, server_count=1):
    total_service_rate = service_rate * server_count
    utilization = arrival_rate / total_service_rate

    if utilization < 0.7:
        status = "여유"
    elif utilization < 1.0:
        status = "혼잡 주의"
    else:
        status = "처리 한계 초과"

    return {
        "도착률": arrival_rate,
        "전체 처리율": total_service_rate,
        "이용률": utilization,
        "상태": status
    }

cases = [
    ("급식대 1개", 18, 24, 1),
    ("급식대 2개", 30, 20, 2),
    ("프린터 1대", 12, 10, 1)
]

for name, arrival, service, servers in cases:
    result = analyze_load(
        arrival_rate=arrival,
        service_rate=service,
        server_count=servers
    )

    print(name)
    print(
        f"이용률: {result['이용률']:.2f}, "
        f"상태: {result['상태']}"
    )
변수의미
arrival_rate단위 시간당 새로 도착하는 학생 또는 작업 수
service_rate서버 하나가 단위 시간당 처리할 수 있는 수
server_count동시에 처리 가능한 배식대·프린터·상담 창구 수
utilization도착률 ÷ 전체 처리율로 계산한 이용률
이용률 1 이상평균적으로 들어오는 양이 처리 능력 이상인 상태
상세 설명
이용률이 1보다 작더라도 1에 가까워질수록 작은 지연이나 처리시간 변화가 긴 대기열로 이어질 수 있습니다. 이 예제는 실제 대기행렬 이론을 단순화한 교육용 모델이며, 개선 전후의 처리 능력을 비교하는 지표로 활용할 수 있습니다.
