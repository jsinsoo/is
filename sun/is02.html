def calculate(a, b, operator="+"):
    if operator == "+":
        return a + b
    elif operator == "-":
        return a - b
    elif operator == "*":
        return a * b
    elif operator == "/":
        return a / b if b != 0 else None

def output_result(result, title="계산 결과"):
    if result is None:
        print("계산할 수 없습니다.")
    else:
        print(f"{title}: {result}")

x, y = input_numbers()
result = calculate(x, y)
output_result(result)

calculate(x, y)처럼 연산자를 생략하면 기본값 "+"가 사용됩니다.

def input_scores():
    scores = list(map(int, input("점수들을 입력하세요: ").split()))

def get_grade(average):
    if average >= 90:
        return "A"
    elif average >= 80:
        return "B"
    elif average >= 70:
        return "C"
    else:
        return "D"

def output_report(average, grade, digits=1):

    print(f"등급: {grade}")

average = calculate_average(scores)
grade = get_grade(average)
output_report(average, grade)
기본 매개변수

예제 7

상품 주문 금액 계산

가격 계산할인율배송비
def input_order():
    price = int(input("상품 가격: "))
    quantity = int(input("수량: "))
    return price, quantity

def calculate_total(price, quantity, discount=0, delivery=3000):
    subtotal = price * quantity
    discount_amount = subtotal * discount
    return subtotal - discount_amount + delivery

def output_total(total, currency="원"):
    print(f"최종 결제 금액: {total:,.0f}{currency}")

price, quantity = input_order()
total = calculate_total(price, quantity)
output_total(total)

활용 예

calculate_total(10000, 3)
기본 할인율 0, 기본 배송비 3000원

calculate_total(10000, 3, 0.1)
10% 할인, 기본 배송비 적용

calculate_total(10000, 3, 0.1, 0)
10% 할인, 무료 배송

def input_text():
    return input("문장: ")

def process_text(text, separator=" ", upper=False):
    words = text.strip().split(separator)
    result = "-".join(words)

    if upper:
        result = result.upper()

    return result

def output_text(result, label="변환 결과"):
    print(f"{label}: {result}")

text = input_text()
result = process_text(text)
output_text(result)
위치 인수 활용

        "개수": len(data),
        "평균": sum(data) / len(data),
        "최댓값": max(data),
        "최솟값": min(data)
    }

    if include_range:
        result["범위"] = max(data) - min(data)

    return result

def output_summary(summary, digits=2):
    for key, value in summary.items():
        if isinstance(value, float):
            print(f"{key}: {value:.{digits}f}")
        else:
            print(f"{key}: {value}")

data = input_data()
summary = summarize(data)
output_summary(summary)

기본 매개변수의 의미

include_range=True이므로 기본적으로 범위를 계산합니다.

summarize(data, False)로 호출하면 범위를 제외할 수 있습니다.

digits=2는 실수 결과를 소수점 둘째 자리까지 표시합니다.

def input_environment():
    temperature = float(input("온도(℃): "))
    humidity = float(input("습도(%): "))
    co2 = int(input("이산화탄소(ppm): "))

def check_temperature(value, minimum=20, maximum=26):
    if minimum <= value <= maximum:
        return "적정"
    return "조절 필요"

def check_humidity(value, minimum=40, maximum=60):
    if minimum <= value <= maximum:
        return "적정"
    return "조절 필요"

def check_co2(value, limit=1000):
    if value < limit:
        return "양호"
    return "환기 필요"

def make_report(temperature, humidity, co2):
    return {
        "온도": check_temperature(temperature),
        "습도": check_humidity(humidity),
        "이산화탄소": check_co2(co2)
    }

def output_report(report, title="교실 환경 진단"):
    print(f"\n[{title}]")
    for item, result in report.items():
        print(f"{item}: {result}")

temperature, humidity, co2 = input_environment()
report = make_report(temperature, humidity, co2)
output_report(report)
