def classify_score(score):
    if not 0 <= score <= 100:
        raise ValueError(
            "점수 범위 오류"
        )

    return "합격" if score >= 60 else "불합격"

tests = [
    (75, "합격"),
    (60, "합격"),
    (59, "불합격"),
    (0, "불합격"),
    (100, "합격")
]

passed = 0

for value, expected in tests:
    actual = classify_score(value)

    if actual == expected:
        passed += 1
        print("PASS", value)
    else:
        print("FAIL", value, actual, expected)

print("통과율:", passed / len(tests) * 100)

예제 3

효율성 평가: 이중 반복문을 집합으로 개선

O(n²)평균 O(n)

개선 전

def has_duplicate_slow(data):
    for i in range(len(data)):
        for j in range(i + 1, len(data)):
            if data[i] == data[j]:
                return True

    return False

개선 후

def has_duplicate_fast(data):
    seen = set()

    for value in data:
        if value in seen:
            return True

        seen.add(value)

    return False
평가
개선 후 시간 복잡도는 평균 O(n)으로 줄지만, 최대 n개의 값을 저장하는 추가 메모리가 필요합니다. 시간과 공간을 함께 평가합니다.

예제 6

편향 평가: 집단별 정확도 비교

편향집단별 성능
results = [
    {"group": "A", "correct": True},
    {"group": "A", "correct": True},
    {"group": "A", "correct": False},
    {"group": "B", "correct": True},
    {"group": "B", "correct": False},
    {"group": "B", "correct": False}
]

stats = {}

for item in results:
    group = item["group"]

    if group not in stats:
        stats[group] = {
            "total": 0,
            "correct": 0
        }

    stats[group]["total"] += 1

    if item["correct"]:
        stats[group]["correct"] += 1

for group, value in stats.items():
    accuracy = (
        value["correct"]
        / value["total"]
    )

    print(
        group,
        "정확도:",
        round(accuracy, 3)
    )
평가 해석
전체 정확도만 보면 특정 집단에서 성능이 낮은 문제를 놓칠 수 있습니다. 표본 수와 데이터 대표성도 함께 확인합니다.

예제 9

위험도와 비용을 반영한 개선 우선순위

우선순위위험도
issues = [
    {
        "name": "개인정보 노출",
        "impact": 5,
        "urgency": 5,
        "effort": 2
    },
    {
        "name": "느린 중복 탐색",
        "impact": 3,
        "urgency": 3,
        "effort": 2
    },
    {
        "name": "출력 문장 불명확",
        "impact": 3,
        "urgency": 4,
        "effort": 1
    }
]

for issue in issues:
    issue["score"] = (
        issue["impact"]
        * issue["urgency"]
        / issue["effort"]
    )

issues.sort(
    key=lambda item: item["score"],
    reverse=True
)

for issue in issues:
    print(
        issue["name"],
        round(issue["score"], 2)
    )

예제 10

종합: 상호 평가 결과와 개선 전후 비교

종합 평가개선 기록
before = {
    "정확성": 3,
    "효율성": 2,
    "사용성": 3,
    "개인정보": 2,
    "편향": 2,
    "저작권": 3
}

after = {
    "정확성": 5,
    "효율성": 4,
    "사용성": 4,
    "개인정보": 5,
    "편향": 4,
    "저작권": 5
}

for category in before:
    change = after[category] - before[category]

    print(
        category,
        before[category],
        "→",
        after[category],
        f"(+{change})"
    )

before_average = (
    sum(before.values())
    / len(before)
)

after_average = (
    sum(after.values())
    / len(after)
)

print("개선 전 평균:", round(before_average, 2))
print("개선 후 평균:", round(after_average, 2))
관점발견 문제개선 내용검증 방법

Ⅶ. 수업 끝 멘트

시는 한 단어를 고치며 마음의 온도를 바꾸고, 소설은 인물의 선택을 다시 바라보며 더 깊은 의미를 만듭니다. 희곡은 서로 다른 해석이 만나 장면을 풍성하게 하고, 연극은 연출가·배우·관객의 피드백 속에서 성장합니다. 음악도 혼자 완성되는 것이 아니라 연주와 감상의 대화를 통해 새로운 울림을 얻습니다.

여러분의 산출물도 처음부터 완벽할 필요는 없습니다. 정확성의 작은 오류, 느린 코드, 어려운 화면, 불필요한 정보 수집, 보이지 않던 편향, 빠진 출처를 발견하는 순간이 바로 더 나은 작품으로 성장하는 출발점입니다.

그리고 동료 평가 점수를 높이려고 친구에게 간식을 제공하면 사용성은 향상될 수 있지만 평가의 공정성은 하락합니다. 오늘 배운 근거 있는 평가와 따뜻한 개선의 태도가 여러분의 코드와 관계, 그리고 꿈을 더 단단하게 만들어 주기를 바랍니다.
