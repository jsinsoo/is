소괄호 짝 검사

def check_parentheses(expression):
    stack = []

    for character in expression:
        if character == "(":
            stack.append(character)

        elif character == ")":
            if not stack:
                return False
            stack.pop()

    return len(stack) == 0

print(check_parentheses("(a+b)*(c-d)"))
print(check_parentheses("((a+b)"))
print(check_parentheses("a+b)"))

판정 원리

def check_brackets(expression):
    stack = []
    pairs = {
        ")": "(",
        "]": "[",
        "}": "{"
    }

    for character in expression:
        if character in "([{":
            stack.append(character)

        elif character in ")]}":
            if not stack:
                return False

            top = stack.pop()

            if top != pairs[character]:
                return False

    return not stack

tests = [
    "{[()]}",
    "{[(])}",
    "(([]))",
    "([)"
]

for test in tests:
    print(test, "→", check_brackets(test))

def diagnose_brackets(expression):
    stack = []
    pairs = {")": "(", "]": "[", "}": "{"}

    for index, character in enumerate(expression):
        if character in "([{":
            stack.append((character, index))

        elif character in ")]}":
            if not stack:
                return f"{index}번 위치: 여는 괄호가 없습니다."

            opening, opening_index = stack.pop()

            if opening != pairs[character]:
                return (
                    f"{index}번 위치: {opening}와 "
                    f"{character}의 종류가 다릅니다."
                )

    if stack:
        opening, index = stack[-1]
        return f"{index}번 위치의 {opening}가 닫히지 않았습니다."

    return "올바른 괄호 구조입니다."

print(diagnose_brackets("{[(])}"))

예제 5

사칙연산 후위 표기식 계산기

사칙연산함수
def evaluate_postfix(expression):
    stack = []

    for token in expression.split():
        if token.lstrip("-").isdigit():
            stack.append(int(token))
            continue

        right = stack.pop()
        left = stack.pop()

        if token == "+":
            result = left + right
        elif token == "-":
            result = left - right
        elif token == "*":
            result = left * right
        elif token == "/":
            result = left / right
        else:
            raise ValueError("지원하지 않는 연산자입니다.")

        stack.append(result)

    if len(stack) != 1:
        raise ValueError("올바르지 않은 후위 표기식입니다.")

    return stack[0]

print(evaluate_postfix("8 2 / 3 +"))
print(evaluate_postfix("5 2 3 * +"))
피연산자 순서
뺄셈과 나눗셈에서는 먼저 꺼낸 값이 오른쪽 피연산자입니다. 반드시 left 연산 right 순서로 계산해야 합니다.

예제 6

후위 표기식 계산 과정 추적

추적표디버깅
def evaluate_with_trace(expression):
    stack = []

    for token in expression.split():
        before = stack.copy()

        if token.isdigit():
            stack.append(int(token))
            action = f"{token} push"

        else:
            right = stack.pop()
            left = stack.pop()

            operations = {
                "+": left + right,
                "-": left - right,
                "*": left * right,
                "/": left / right
            }

            stack.append(operations[token])
            action = f"{left} {token} {right}"

        print(
            f"토큰={token:>2} | 이전={before} "
            f"| 처리={action} | 이후={stack}"
        )

    return stack.pop()

print("결과:", evaluate_with_trace("7 2 3 * -"))
학습 효과
각 토큰 처리 전후의 스택을 출력하면 잘못된 피연산자 순서와 스택 부족 오류를 쉽게 찾을 수 있습니다.

예제 7

중위 표기식을 후위 표기식으로 변환

연산자 스택우선순위
def infix_to_postfix(expression):
    output = []
    operators = []
    priority = {"+": 1, "-": 1, "*": 2, "/": 2}

    for token in expression.split():
        if token.isdigit():
            output.append(token)

        elif token == "(":
            operators.append(token)

        elif token == ")":
            while operators and operators[-1] != "(":
                output.append(operators.pop())
            operators.pop()

        else:
            while (
                operators
                and operators[-1] != "("
                and priority[operators[-1]] >= priority[token]
            ):
                output.append(operators.pop())

            operators.append(token)

    while operators:
        output.append(operators.pop())

    return " ".join(output)

print(infix_to_postfix("2 + 3 * 4"))
print(infix_to_postfix("( 2 + 3 ) * 4"))
스택 역할
연산자는 바로 출력하지 않고 스택에 저장하며, 우선순위가 높은 연산자부터 꺼내 후위식에 추가합니다.

예제 9

실행 취소와 다시 실행

UndoRedo두 스택
document = ""
undo_stack = []
redo_stack = []

def write(text):
    global document

    undo_stack.append(document)
    document += text
    redo_stack.clear()

def undo():
    global document

    if undo_stack:
        redo_stack.append(document)
        document = undo_stack.pop()

def redo():
    global document

    if redo_stack:
        undo_stack.append(document)
        document = redo_stack.pop()

write("파이썬")
write(" 스택")
write(" 실습")

print(document)
undo()
print("undo:", document)
undo()
print("undo:", document)
redo()
print("redo:", document)
두 스택의 역할
Undo 스택은 이전 상태를, Redo 스택은 취소된 이후 상태를 저장합니다. 새 작업을 수행하면 Redo 기록을 비웁니다.

예제 10

명령어 기반 간단한 문서 편집기

종합 프로그램상태 관리
document = ""
undo_stack = []
redo_stack = []

def show():
    print(f"현재 문서: [{document}]")

while True:
    command = input("명령: ").split(maxsplit=1)
    action = command[0]

    if action == "write":
        if len(command) == 1:
            print("추가할 내용을 입력하세요.")
            continue

        undo_stack.append(document)
        document += command[1]
        redo_stack.clear()
        show()

    elif action == "undo":
        if undo_stack:
            redo_stack.append(document)
            document = undo_stack.pop()
        else:
            print("취소할 작업이 없습니다.")
        show()

    elif action == "redo":
        if redo_stack:
            undo_stack.append(document)
            document = redo_stack.pop()
        else:
            print("다시 실행할 작업이 없습니다.")
        show()

    elif action == "show":
        show()

    elif action == "quit":
        break

    else:
        print("알 수 없는 명령입니다.")

Ⅶ. 수업 끝 멘트

“시는 열린 마음에 마지막 구절로 조용히 괄호를 닫고, 소설은 앞선 사건의 기억을 다음 장면에서 다시 꺼냅니다. 희곡은 대사의 순서가 어긋나면 장면이 흔들리지만, 연극은 배우들이 서로의 마지막 말을 기억하며 무대를 이어 갑니다. 음악도 긴장된 화음이 다음 화음에서 풀릴 때 비로소 이야기가 완성됩니다. 여러분의 꿈에도 열어 둔 괄호가 있고 아직 계산하지 않은 식이 있습니다. 결과가 마음에 들지 않으면 Undo로 배우되, 계속 되돌아가기만 하지 말고 언젠가는 Redo보다 더 멋진 새 작업을 시작하세요. 단, 인생의 시험 점수에는 Ctrl+Z가 없으니 오늘의 연습 스택을 충분히 쌓아 두는 것이 좋습니다.”
