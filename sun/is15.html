from collections import deque

def is_palindrome(text):
    characters = deque(
        character.lower()
        for character in text
        if character.isalnum()
    )

    while len(characters) > 1:
        left = characters.popleft()
        right = characters.pop()

        if left != right:
            return False

    return True

print(is_palindrome("level"))
print(is_palindrome("python"))
선택 근거
문자열의 왼쪽과 오른쪽을 반복해서 제거해야 하므로 양쪽 끝 연산이 빠른 덱이 적합합니다.

예제 9

교실 이동의 최소 단계 — 그래프와 BFS 선택

그래프BFS큐
from collections import deque

building = {
    "교실": ["복도"],
    "복도": ["교실", "계단", "도서관"],
    "계단": ["복도", "과학실"],
    "도서관": ["복도", "과학실"],
    "과학실": ["계단", "도서관"]
}

def shortest_path(start, target):
    queue = deque([(start, [start])])
    visited = {start}

    while queue:
        place, path = queue.popleft()

        if place == target:
            return path

        for next_place in building[place]:
            if next_place not in visited:
                visited.add(next_place)
                queue.append(
                    (next_place, path + [next_place])
                )

    return None

path = shortest_path("교실", "과학실")

print("경로:", path)
print("이동 단계:", len(path) - 1)
선택 근거
장소 연결은 그래프로 나타내고, 모든 이동 비용이 같을 때 최소 간선 수 경로는 큐 기반 BFS로 찾습니다.

def recommend_structure(
    keep_order=False,
    remove_duplicates=False,
    key_lookup=False,
    recent_first=False,
    arrival_first=False,
    both_ends=False,
    priority=False,
    relationships=False
):
    if relationships:
        return "그래프", "대상 사이의 연결과 경로를 표현할 수 있습니다."

    if priority:
        return "힙", "최우선 자료를 반복해서 효율적으로 꺼낼 수 있습니다."

    if recent_first:
        return "스택", "가장 최근 자료를 먼저 처리하는 LIFO 구조입니다."

    if arrival_first:
        return "큐", "먼저 들어온 자료를 먼저 처리하는 FIFO 구조입니다."

    if both_ends:
        return "덱", "양쪽 끝의 삽입과 삭제가 모두 효율적입니다."

    if remove_duplicates:
        return "집합", "중복을 자동으로 제거하고 포함 여부를 빠르게 확인합니다."

    if key_lookup:
        return "딕셔너리", "키를 이용해 값을 빠르게 조회하고 갱신합니다."

    if keep_order:
        return "리스트", "자료의 순서와 인덱스를 유지합니다."
