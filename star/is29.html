def average_waiting(service_times):
    current = 0
    waits = []

    for service in service_times:
        waits.append(current)
        current += service

    return sum(waits) / len(waits)

times = [2, 3, 2, 4, 1]
print("평균 대기시간:", average_waiting(times))

from collections import deque

def shortest_path(graph, start, target):
    queue = deque([(start, [start])])
    visited = {start}

    while queue:
        node, path = queue.popleft()

        if node == target:
            return path

        for neighbor in graph[node]:
            if neighbor not in visited:
                visited.add(neighbor)
                queue.append((neighbor, path + [neighbor]))

    return None

일정

지도 조사 2일 → 그래프 설계 2일 → BFS 구현 2일 → 사용자 테스트 2일

selected = []

def arrange():
    if len(selected) == len(students):
        print(selected)
        return

    for student in students:
        if student in selected:
            continue

        selected.append(student)
        arrange()
        selected.pop()

마무리 진단 문제

1. 프로젝트의 사용자를 먼저 정의해야 하는 이유는 무엇인가?
정답 보기사용자의 불편과 사용 상황을 기준으로 필요한 기능과 출력 형태를 정하기 위해서입니다.
2. 입력과 출력 정의에 포함해야 할 내용은 무엇인가?
정답 보기입력 자료의 종류·형식·범위와 출력 결과의 내용·형식·제공 방법을 포함해야 합니다.
3. 기능 분해의 장점은 무엇인가?
정답 보기복잡한 프로젝트를 작은 작업으로 나누어 구현·분담·테스트하기 쉬워집니다.
4. 팀 역할을 나눌 때 중요한 기준은 무엇인가?
정답 보기각 역할의 책임과 산출물, 완료 기준이 겹치거나 빠지지 않도록 구체적으로 정해야 합니다.
5. 수행 계획의 완료 기준이 필요한 이유는 무엇인가?
정답 보기단순히 작업을 했다는 사실이 아니라 목표 수준에 도달했는지 객관적으로 판단하기 위해서입니다.
