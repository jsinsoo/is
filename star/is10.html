예제 1

소괄호 짝 검사

괄호 검사LIFO
def check_parentheses(expression):
    stack = []

    for character in expression:
        if character == "(":
            stack.append(character)

        elif character == ")":
            if not stack:
                return False
            stack.pop()

    return len(stack) == 0

print(check_parentheses("(a+b)*(c-d)"))
print(check_parentheses("((a+b)"))
print(check_parentheses("a+b)"))

판정 원리

여는 괄호는 스택에 넣고 닫는 괄호를 만나면 하나를 꺼냅니다.

닫는 괄호가 남거나 마지막에 여는 괄호가 남으면 잘못된 식입니다.

예제 2

소·중·대괄호 검사

괄호 종류딕셔너리
def check_brackets(expression):
    stack = []
    pairs = {
        ")": "(",
        "]": "[",
        "}": "{"
    }

    for character in expression:
        if character in "([{":
            stack.append(character)

        elif character in ")]}":
            if not stack:
                return False

            top = stack.pop()

            if top != pairs[character]:
                return False

    return not stack

tests = [
    "{[()]}",
    "{[(])}",
    "(([]))",
    "([)"
]

for test in tests:
    print(test, "→", check_brackets(test))
LIFO 필요성
가장 나중에 열린 괄호가 가장 먼저 닫혀야 하므로 스택이 적합합니다.
예제 3

괄호 오류 위치 알려 주기

오류 진단인덱스 저장
def diagnose_brackets(expression):
    stack = []
    pairs = {")": "(", "]": "[", "}": "{"}

    for index, character in enumerate(expression):
        if character in "([{":
            stack.append((character, index))

        elif character in ")]}":
            if not stack:
                return f"{index}번 위치: 여는 괄호가 없습니다."

            opening, opening_index = stack.pop()

            if opening != pairs[character]:
                return (
                    f"{index}번 위치: {opening}와 "
                    f"{character}의 종류가 다릅니다."
                )

    if stack:
        opening, index = stack[-1]
        return f"{index}번 위치의 {opening}가 닫히지 않았습니다."

    return "올바른 괄호 구조입니다."

print(diagnose_brackets("{[(])}"))
확장
괄호 문자와 위치를 튜플로 저장하면 단순한 참·거짓 판정을 넘어 오류 위치까지 안내할 수 있습니다.
예제 4

후위 표기식의 스택 변화

후위 표기식피연산자
tokens = "2 3 + 4 *".split()
stack = []

for token in tokens:
    if token.isdigit():
        stack.append(int(token))
        print("숫자 push:", stack)

    else:
        right = stack.pop()
        left = stack.pop()
        result = left + right if token == "+" else left * right
        stack.append(result)
        print("연산 후:", stack)

print("최종 결과:", stack.pop())

계산 순서

2 push
3 push
+ → 2+3 = 5
4 push
* → 5×4 = 20

후위식에는 괄호와 연산자 우선순위 처리가 필요하지 않습니다.

예제 5

사칙연산 후위 표기식 계산기

사칙연산함수
def evaluate_postfix(expression):
    stack = []

    for token in expression.split():
        if token.lstrip("-").isdigit():
            stack.append(int(token))
            continue

        right = stack.pop()
        left = stack.pop()

        if token == "+":
            result = left + right
        elif token == "-":
            result = left - right
        elif token == "*":
            result = left * right
        elif token == "/":
            result = left / right
        else:
            raise ValueError("지원하지 않는 연산자입니다.")

        stack.append(result)

    if len(stack) != 1:
        raise ValueError("올바르지 않은 후위 표기식입니다.")

    return stack[0]

print(evaluate_postfix("8 2 / 3 +"))
print(evaluate_postfix("5 2 3 * +"))
피연산자 순서
뺄셈과 나눗셈에서는 먼저 꺼낸 값이 오른쪽 피연산자입니다. 반드시 left 연산 right 순서로 계산해야 합니다.
예제 6

후위 표기식 계산 과정 추적

추적표디버깅
def evaluate_with_trace(expression):
    stack = []

    for token in expression.split():
        before = stack.copy()

        if token.isdigit():
            stack.append(int(token))
            action = f"{token} push"

        else:
            right = stack.pop()
            left = stack.pop()

            operations = {
                "+": left + right,
                "-": left - right,
                "*": left * right,
                "/": left / right
            }

            stack.append(operations[token])
            action = f"{left} {token} {right}"

        print(
            f"토큰={token:>2} | 이전={before} "
            f"| 처리={action} | 이후={stack}"
        )

    return stack.pop()

print("결과:", evaluate_with_trace("7 2 3 * -"))
학습 효과
각 토큰 처리 전후의 스택을 출력하면 잘못된 피연산자 순서와 스택 부족 오류를 쉽게 찾을 수 있습니다.
예제 7

중위 표기식을 후위 표기식으로 변환

연산자 스택우선순위
def infix_to_postfix(expression):
    output = []
    operators = []
    priority = {"+": 1, "-": 1, "*": 2, "/": 2}

    for token in expression.split():
        if token.isdigit():
            output.append(token)

        elif token == "(":
            operators.append(token)

        elif token == ")":
            while operators and operators[-1] != "(":
                output.append(operators.pop())
            operators.pop()

        else:
            while (
                operators
                and operators[-1] != "("
                and priority[operators[-1]] >= priority[token]
            ):
                output.append(operators.pop())

            operators.append(token)

    while operators:
        output.append(operators.pop())

    return " ".join(output)

print(infix_to_postfix("2 + 3 * 4"))
print(infix_to_postfix("( 2 + 3 ) * 4"))
스택 역할
연산자는 바로 출력하지 않고 스택에 저장하며, 우선순위가 높은 연산자부터 꺼내 후위식에 추가합니다.

document = ""
undo_stack = []

def write(text):
    global document

    undo_stack.append(document)
    document += text
    print("문서:", document)

def undo():
    global document

    if not undo_stack:
        print("취소할 작업이 없습니다.")
        return

    document = undo_stack.pop()
    print("취소 후:", document)

write("정보")
write(" 수업")
write(" 자료")
undo()
undo()
핵심
문서를 변경하기 전에 현재 상태를 push하고, 실행 취소할 때 가장 최근 상태를 pop하여 복원합니다.
예제 9

실행 취소와 다시 실행

UndoRedo두 스택
document = ""
undo_stack = []
redo_stack = []

def write(text):
    global document

    undo_stack.append(document)
    document += text
    redo_stack.clear()

def undo():
    global document

    if undo_stack:
        redo_stack.append(document)
        document = undo_stack.pop()

def redo():
    global document

    if redo_stack:
        undo_stack.append(document)
        document = redo_stack.pop()

write("파이썬")
write(" 스택")
write(" 실습")

print(document)
undo()
print("undo:", document)
undo()
print("undo:", document)
redo()
print("redo:", document)
두 스택의 역할
Undo 스택은 이전 상태를, Redo 스택은 취소된 이후 상태를 저장합니다. 새 작업을 수행하면 Redo 기록을 비웁니다.
예제 10

명령어 기반 간단한 문서 편집기

종합 프로그램상태 관리
document = ""
undo_stack = []
redo_stack = []

def show():
    print(f"현재 문서: [{document}]")

while True:
    command = input("명령: ").split(maxsplit=1)
    action = command[0]

    if action == "write":
        if len(command) == 1:
            print("추가할 내용을 입력하세요.")
            continue

        undo_stack.append(document)
        document += command[1]
        redo_stack.clear()
        show()

    elif action == "undo":
        if undo_stack:
            redo_stack.append(document)
            document = undo_stack.pop()
        else:
            print("취소할 작업이 없습니다.")
        show()

    elif action == "redo":
        if redo_stack:
            undo_stack.append(document)
            document = redo_stack.pop()
        else:
            print("다시 실행할 작업이 없습니다.")
        show()

    elif action == "show":
        show()

    elif action == "quit":
        break

    else:
        print("알 수 없는 명령입니다.")
명령기능예시
write 내용문서 뒤에 내용 추가write 안녕하세요
undo최근 작업 취소undo
redo취소한 작업 다시 실행redo
show현재 문서 출력show
quit프로그램 종료quit

마무리 진단 문제

1. 괄호 검사에서 여는 괄호를 스택에 넣는 이유는 무엇인가?
정답 보기가장 최근에 열린 괄호와 다음 닫는 괄호의 짝을 확인하기 위해서입니다.
2. 후위 표기식에서 연산자를 만났을 때 몇 개의 피연산자를 꺼내는가?
정답 보기이항 연산자는 두 개의 피연산자를 꺼냅니다.
3. 뺄셈 계산에서 스택에서 먼저 꺼낸 값은 왼쪽인가 오른쪽 피연산자인가?
정답 보기오른쪽 피연산자입니다.
4. 실행 취소에서 문서를 변경하기 전에 저장해야 하는 것은 무엇인가?
정답 보기변경 전의 문서 상태입니다.
5. 새로운 작업을 수행했을 때 Redo 스택을 비우는 이유는 무엇인가?
정답 보기새 작업으로 작업 이력이 갈라지므로 이전에 취소했던 미래 상태를 더 이상 그대로 다시 실행할 수 없기 때문입니다.
