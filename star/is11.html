1. FIFO와 원형 큐

FIFO먼저 들어온 자료가 먼저 나오는 선입선출 방식입니다.
enqueue큐의 뒤쪽에 새로운 자료를 추가합니다.
dequeue큐의 앞쪽 자료를 제거하고 반환합니다.
원형 큐배열의 끝과 처음을 연결해 빈 공간을 다시 사용합니다.
일반 배열 큐는 rear가 배열 끝에 도달하면 앞쪽에 빈 공간이 있어도 새 자료를 넣기 어렵습니다. 원형 큐는 (index + 1) % capacity 계산으로 인덱스를 처음 위치로 되돌려 공간을 재사용합니다.
연산deque 구현직접 구현 의미
enqueue(x)queue.append(x)rear 위치에 자료 저장
dequeue()queue.popleft()front 위치 자료 반환
peek()queue[0]front 자료 확인
인덱스 순환내부에서 자동 처리(index+1)%capacity

예제 2

리스트로 enqueue·dequeue 직접 구현

enqueuedequeue
queue = []

def enqueue(value):
    queue.append(value)
    print(f"enqueue({value}) → {queue}")

def dequeue():
    if not queue:
        return None

    value = queue[0]
    del queue[0]
    return value

enqueue(10)
enqueue(20)
enqueue(30)

print("dequeue:", dequeue())
print("현재 큐:", queue)
원리
삽입은 리스트 뒤에서, 삭제는 리스트 앞에서 수행하므로 FIFO 순서가 유지됩니다.

예제 4

enqueue·dequeue·peek 함수 구현

함수 분리예외 처리
from collections import deque

queue = deque()

def enqueue(value):
    queue.append(value)

def dequeue():
    if not queue:
        return "EMPTY"
    return queue.popleft()

def peek():
    if not queue:
        return "EMPTY"
    return queue[0]

def size():
    return len(queue)

enqueue("A")
enqueue("B")
enqueue("C")

print("peek:", peek())
print("dequeue:", dequeue())
print("크기:", size())
peek
맨 앞 자료를 삭제하지 않고 확인하므로 호출 후에도 큐의 내용은 그대로 유지됩니다.
예제 5

번호표 대기 시스템

실생활 응용자동 번호
from collections import deque

waiting = deque()
next_number = 1

def issue_ticket(name):
    global next_number

    waiting.append((next_number, name))
    print(f"{name}님의 번호는 {next_number}번입니다.")
    next_number += 1

def call_next():
    if not waiting:
        print("대기자가 없습니다.")
        return

    number, name = waiting.popleft()
    print(f"{number}번 {name}님, 입장하세요.")

issue_ticket("민준")
issue_ticket("서연")
issue_ticket("지우")

call_next()
call_next()
FIFO 적용
번호표를 먼저 받은 사람이 먼저 호출되므로 큐의 처리 원리와 정확히 일치합니다.
예제 6

선형 배열 큐의 공간 낭비

frontrear한계 분석
capacity = 5
queue = [None] * capacity
front = 0
rear = 0

def enqueue(value):
    global rear

    if rear == capacity:
        print("rear가 끝에 도달했습니다.")
        return

    queue[rear] = value
    rear += 1

def dequeue():
    global front

    if front == rear:
        return None

    value = queue[front]
    queue[front] = None
    front += 1
    return value

for value in [10, 20, 30, 40, 50]:
    enqueue(value)

print(dequeue())
print(dequeue())
print("배열:", queue)

enqueue(60)
문제점
앞쪽 두 칸이 비어 있지만 rear가 배열 끝에 도달했으므로 60을 삽입하지 못합니다. 이 공간 낭비를 해결하는 구조가 원형 큐입니다.

예제 8

원형 큐 클래스 직접 구현

원형 배열클래스
class CircularQueue:
    def __init__(self, capacity):
        self.capacity = capacity
        self.data = [None] * capacity
        self.front = 0
        self.rear = 0
        self.count = 0

    def is_empty(self):
        return self.count == 0

    def is_full(self):
        return self.count == self.capacity

    def enqueue(self, value):
        if self.is_full():
            raise OverflowError("큐가 가득 찼습니다.")

        self.data[self.rear] = value
        self.rear = (self.rear + 1) % self.capacity
        self.count += 1

    def dequeue(self):
        if self.is_empty():
            raise IndexError("큐가 비어 있습니다.")

        value = self.data[self.front]
        self.data[self.front] = None
        self.front = (self.front + 1) % self.capacity
        self.count -= 1
        return value

    def peek(self):
        if self.is_empty():
            return None
        return self.data[self.front]

queue = CircularQueue(4)

queue.enqueue("A")
queue.enqueue("B")
queue.enqueue("C")

print(queue.dequeue())
queue.enqueue("D")
queue.enqueue("E")

print(queue.data)
print("front:", queue.front)
print("rear:", queue.rear)
count의 역할
front와 rear가 같은 경우 빈 상태인지 가득 찬 상태인지 구분하기 위해 저장된 자료 수를 별도로 관리합니다.
예제 9

원형 큐 상태 변화 추적

상태 추적디버깅
class CircularQueue:
    def __init__(self, capacity):
        self.data = [None] * capacity
        self.capacity = capacity
        self.front = 0
        self.rear = 0
        self.count = 0

    def show(self, action):
        print(
            f"{action:12} | data={self.data} "
            f"| front={self.front} "
            f"| rear={self.rear} "
            f"| count={self.count}"
        )

    def enqueue(self, value):
        if self.count == self.capacity:
            self.show("enqueue 실패")
            return

        self.data[self.rear] = value
        self.rear = (self.rear + 1) % self.capacity
        self.count += 1
        self.show(f"enqueue {value}")

    def dequeue(self):
        if self.count == 0:
            self.show("dequeue 실패")
            return None

        value = self.data[self.front]
        self.data[self.front] = None
        self.front = (self.front + 1) % self.capacity
        self.count -= 1
        self.show(f"dequeue {value}")
        return value

q = CircularQueue(4)

q.enqueue(10)
q.enqueue(20)
q.enqueue(30)
q.dequeue()
q.dequeue()
q.enqueue(40)
q.enqueue(50)
q.enqueue(60)
관찰
rear가 배열 끝을 지나 0과 1 위치로 돌아가, 앞에서 비워진 공간을 다시 사용하는 과정을 확인할 수 있습니다.
예제 10

원형 큐 기반 프린터 작업 스케줄러

종합 프로그램작업 처리
class PrintQueue:
    def __init__(self, capacity=5):
        self.capacity = capacity
        self.data = [None] * capacity
        self.front = 0
        self.rear = 0
        self.count = 0

    def add_job(self, document):
        if self.count == self.capacity:
            print("대기열이 가득 찼습니다.")
            return

        self.data[self.rear] = document
        self.rear = (self.rear + 1) % self.capacity
        self.count += 1
        print("추가:", document)

    def print_next(self):
        if self.count == 0:
            print("인쇄할 문서가 없습니다.")
            return

        document = self.data[self.front]
        self.data[self.front] = None
        self.front = (self.front + 1) % self.capacity
        self.count -= 1
        print("인쇄:", document)

    def show_jobs(self):
        jobs = []

        for offset in range(self.count):
            index = (self.front + offset) % self.capacity
            jobs.append(self.data[index])

        print("대기 문서:", jobs)

printer = PrintQueue(4)

printer.add_job("수업자료.pdf")
printer.add_job("평가계획.hwp")
printer.add_job("학생명단.xlsx")

printer.print_next()
printer.add_job("알고리즘.pdf")
printer.add_job("과제안내.txt")

printer.show_jobs()
설계 요소구현 내용
FIFO먼저 등록된 문서를 먼저 인쇄
원형 인덱스front와 rear를 나머지 연산으로 순환
가득 찬 상태count와 capacity가 같을 때 삽입 제한
빈 상태count가 0일 때 삭제 제한
논리적 순서 출력front부터 count개만 순환하며 확인

마무리 진단 문제

1. FIFO는 어떤 자료 처리 순서를 의미하는가?
정답 보기먼저 들어온 자료를 먼저 처리하는 선입선출 순서입니다.
2. deque에서 큐의 앞 자료를 제거하는 메서드는 무엇인가?
정답 보기popleft()입니다.
3. 선형 배열 큐에서 앞쪽 빈 공간을 다시 사용하기 어려운 이유는 무엇인가?
정답 보기rear 인덱스가 배열 끝에 도달하면 앞쪽이 비어 있어도 뒤로 더 이동할 수 없기 때문입니다.
4. 원형 큐에서 다음 인덱스를 계산하는 식은 무엇인가?
정답 보기(현재 인덱스 + 1) % capacity입니다.
5. front와 rear가 같을 때 빈 큐와 가득 찬 큐를 구분하는 방법은 무엇인가?
정답 보기저장된 자료 수인 count를 관리하거나 한 칸을 항상 비워 두는 방식을 사용할 수 있습니다.
