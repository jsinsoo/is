예제 1

새 리스트를 만드는 기초 퀵 정렬

피벗분할 정복
def quick_sort(data):
    if len(data) <= 1:
        return data

    pivot = data[0]

    left = [
        value for value in data[1:]
        if value <= pivot
    ]

    right = [
        value for value in data[1:]
        if value > pivot
    ]

    return (
        quick_sort(left)
        + [pivot]
        + quick_sort(right)
    )

numbers = [7, 3, 8, 2, 5, 1]
print(quick_sort(numbers))

구조

[7,3,8,2,5,1]
피벗: 7
왼쪽: [3,2,5,1]
오른쪽: [8]

정렬된 왼쪽 + [7] + 정렬된 오른쪽
예제 2

퀵 정렬 분할·결합 과정 추적

호출 깊이중간 결과
def quick_sort_trace(data, depth=0):
    indent = "  " * depth
    print(f"{indent}입력: {data}")

    if len(data) <= 1:
        print(f"{indent}기본 상태: {data}")
        return data

    pivot = data[0]
    left = [x for x in data[1:] if x <= pivot]
    right = [x for x in data[1:] if x > pivot]

    print(
        f"{indent}피벗={pivot}, "
        f"왼쪽={left}, 오른쪽={right}"
    )

    left_sorted = quick_sort_trace(
        left, depth + 1
    )
    right_sorted = quick_sort_trace(
        right, depth + 1
    )

    result = left_sorted + [pivot] + right_sorted
    print(f"{indent}결합: {result}")

    return result

quick_sort_trace([5, 2, 7, 1, 4])
관찰
분할할 때는 피벗을 기준으로 두 부분을 만들고, 재귀 호출이 끝날 때 정렬된 부분들을 결합합니다.
예제 3

제자리 퀵 정렬 직접 구현

Lomuto 분할교환
def partition(data, low, high):
    pivot = data[high]
    smaller = low - 1

    for index in range(low, high):
        if data[index] <= pivot:
            smaller += 1
            data[smaller], data[index] = (
                data[index], data[smaller]
            )

    data[smaller + 1], data[high] = (
        data[high], data[smaller + 1]
    )

    return smaller + 1

def quick_sort(data, low, high):
    if low >= high:
        return

    pivot_index = partition(data, low, high)

    quick_sort(data, low, pivot_index - 1)
    quick_sort(data, pivot_index + 1, high)

numbers = [9, 4, 7, 3, 10, 5]
quick_sort(numbers, 0, len(numbers) - 1)

print(numbers)
결합 방식
제자리 퀵 정렬은 분할 과정에서 피벗의 최종 위치가 결정되므로 별도의 리스트 결합이 필요하지 않습니다.
예제 4

피벗 선택에 따른 분할 균형 비교

첫 값중앙값최악 경우
def split_sizes(data, pivot):
    left = [x for x in data if x < pivot]
    equal = [x for x in data if x == pivot]
    right = [x for x in data if x > pivot]

    return len(left), len(equal), len(right)

data = list(range(1, 16))

first_pivot = data[0]
middle_pivot = data[len(data) // 2]

print(
    "첫 값 피벗:",
    first_pivot,
    split_sizes(data, first_pivot)
)

print(
    "중앙값 피벗:",
    middle_pivot,
    split_sizes(data, middle_pivot)
)
효율성
이미 정렬된 자료에서 첫 값을 피벗으로 선택하면 한쪽 부분이 거의 비어 O(n²)에 가까워질 수 있습니다. 균형 있게 나뉘면 평균적으로 O(n log n)입니다.
예제 5

기본 재귀 거듭제곱

지수 n-1기초 재귀
def power_basic(base, exponent):
    if exponent == 0:
        return 1

    return base * power_basic(
        base,
        exponent - 1
    )

print(power_basic(2, 10))

관계

P(a, 0) = 1
P(a, n) = a × P(a, n-1)

10 → 9 → 8 → ... → 0

지수를 한 번에 1씩 줄이므로 n번 정도의 곱셈이 필요합니다.

예제 6

분할 정복을 이용한 빠른 거듭제곱

지수 절반제곱 결합
def fast_power(base, exponent):
    if exponent == 0:
        return 1

    half = fast_power(
        base,
        exponent // 2
    )

    if exponent % 2 == 0:
        return half * half

    return base * half * half

print(fast_power(2, 13))

짝수 지수

aⁿ = (aⁿᐟ²)²

홀수 지수

aⁿ = a × (a⌊ⁿᐟ²⌋)²
크기 변화
13→6→3→1→0처럼 지수가 절반씩 줄어 호출 단계가 크게 감소합니다.
예제 7

반복문 기반 이진 거듭제곱

비트반복 방식
def power_iterative(base, exponent):
    result = 1

    while exponent > 0:
        if exponent % 2 == 1:
            result *= base

        base *= base
        exponent //= 2

    return result

print(power_iterative(3, 13))
원리
지수의 이진 표현에서 1인 자리의 거듭제곱만 결과에 곱합니다. 재귀 호출 스택을 사용하지 않으면서 O(log n) 단계로 계산합니다.
예제 8

두 점 사이 거리 함수

좌표유클리드 거리
from math import sqrt

def distance(point1, point2):
    x1, y1 = point1
    x2, y2 = point2

    return sqrt(
        (x2 - x1) ** 2
        + (y2 - y1) ** 2
    )

a = (1, 2)
b = (4, 6)

print(distance(a, b))

계산 과정

가로 차이: 4 - 1 = 3
세로 차이: 6 - 2 = 4
거리: √(3² + 4²) = 5
예제 9

가장 가까운 두 점 완전 탐색

모든 점 쌍기초 수준
from math import hypot

def closest_pair_brute(points):
    minimum = float("inf")
    closest = None
    comparisons = 0

    for i in range(len(points)):
        for j in range(i + 1, len(points)):
            comparisons += 1

            x1, y1 = points[i]
            x2, y2 = points[j]

            current = hypot(
                x2 - x1,
                y2 - y1
            )

            if current < minimum:
                minimum = current
                closest = (
                    points[i],
                    points[j]
                )

    return closest, minimum, comparisons

points = [
    (1, 2), (5, 5), (2, 3),
    (8, 8), (6, 5), (3, 1)
]

pair, distance_value, count = (
    closest_pair_brute(points)
)

print("가장 가까운 점:", pair)
print("거리:", distance_value)
print("비교 횟수:", count)
효율성
점이 n개이면 모든 두 점 조합을 비교하므로 비교 횟수는 n(n-1)/2이고 시간 복잡도는 O(n²)입니다.
예제 10

가장 가까운 두 점 분할 정복 확장

심화중앙 띠분할 정복
from math import hypot

def point_distance(a, b):
    return hypot(
        a[0] - b[0],
        a[1] - b[1]
    )

def brute_force(points):
    minimum = float("inf")
    pair = None

    for i in range(len(points)):
        for j in range(i + 1, len(points)):
            current = point_distance(
                points[i],
                points[j]
            )

            if current < minimum:
                minimum = current
                pair = (points[i], points[j])

    return minimum, pair

def closest_pair(points):
    points = sorted(points)

    def solve(px):
        n = len(px)

        if n <= 3:
            return brute_force(px)

        middle = n // 2
        middle_x = px[middle][0]

        left_distance, left_pair = solve(
            px[:middle]
        )
        right_distance, right_pair = solve(
            px[middle:]
        )

        if left_distance <= right_distance:
            minimum = left_distance
            best_pair = left_pair
        else:
            minimum = right_distance
            best_pair = right_pair

        strip = [
            point for point in px
            if abs(point[0] - middle_x) < minimum
        ]

        strip.sort(key=lambda point: point[1])

        for i in range(len(strip)):
            for j in range(
                i + 1,
                min(i + 8, len(strip))
            ):
                current = point_distance(
                    strip[i],
                    strip[j]
                )

                if current < minimum:
                    minimum = current
                    best_pair = (
                        strip[i],
                        strip[j]
                    )

        return minimum, best_pair

    return solve(points)

points = [
    (1, 2), (5, 5), (2, 3),
    (8, 8), (6, 5), (3, 1),
    (4, 4.2), (4.1, 4.3)
]

minimum, pair = closest_pair(points)

print("가장 가까운 두 점:", pair)
print("최소 거리:", minimum)
단계가장 가까운 두 점 문제의 처리
분할x좌표 순으로 정렬한 점을 왼쪽과 오른쪽 절반으로 나눔
해결각 절반에서 가장 가까운 두 점을 재귀적으로 계산
부분 결과왼쪽·오른쪽 최소 거리 중 작은 값을 선택
결합중앙선 주변의 띠 영역에서 양쪽에 걸친 더 가까운 점 쌍 검사
효율성정교하게 구현하면 O(n log n), 예제는 이해 중심의 단순화 버전
수준별 선택 설명
“기초 단계에서는 코드 구조가 명확한 퀵 정렬과 완전 탐색 방식의 두 점 거리 문제를 구현한다. 중급 단계에서는 빠른 거듭제곱으로 입력 크기를 절반씩 줄이는 원리를 확인한다. 심화 단계에서는 가장 가까운 두 점 문제를 좌우 부분 문제와 중앙 띠 문제로 분해하여 해결한다.”

마무리 진단 문제

1. 퀵 정렬에서 배열을 나누는 기준값을 무엇이라 하는가?
정답 보기피벗입니다.
2. 퀵 정렬에서 피벗 선택이 중요한 이유는 무엇인가?
정답 보기피벗에 따라 부분 배열의 균형이 달라져 실행 시간이 O(n log n)에서 O(n²)까지 달라질 수 있기 때문입니다.
3. 빠른 거듭제곱에서 지수는 호출할 때마다 어떻게 변하는가?
정답 보기정수 나눗셈을 이용해 대략 절반으로 줄어듭니다.
4. 가장 가까운 두 점 완전 탐색의 시간 복잡도는 무엇인가?
정답 보기모든 두 점 쌍을 비교하므로 O(n²)입니다.
5. 가장 가까운 두 점 분할 정복에서 결합 단계에 확인하는 영역은 어디인가?
정답 보기왼쪽과 오른쪽 경계에 걸친 더 가까운 점을 찾기 위해 중앙선 주변의 띠 영역을 확인합니다.
